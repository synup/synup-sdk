Synup SDK

Build local presence automations in minutes — review auto-responders, listings monitors, reputation dashboards, and more.

Synup SDK v0.4.0
What you can build
  • Auto-reply to reviews across all locations
  • Bulk onboard 500+ locations from CSV
  • Monitor listing sync health across 80+ directories
  • Generate weekly reputation reports
  • Launch SMS/email review campaigns

Create account & get API key

  1. Sign up at synup.dev
  2. Go to Settings → Integrations → Generate
  3. Copy your API key and set it as an environment variable:
export SYNUP_API_KEY="your-api-key-here"
export SYNUP_API_KEY="your-api-key-here"

Quickstart

Auto-reply to positive reviews in 10 lines. Set SYNUP_API_KEY in your environment and run:

import synup

client = synup.Synup()  # reads SYNUP_API_KEY from env

# Auto-reply to all 4-5 star reviews that haven't been answered
results = client.workflows.auto_reply_to_reviews(
    16808,
    template="Thanks for the {rating}-star review!",
    min_rating=4,
)
print(f"Replied to {len(results)} reviews")
import { SynupClient } from 'synup-js'

const client = new SynupClient({ apiKey: process.env.SYNUP_API_KEY })

// Fetch recent reviews and reply to 4-5 star ones
const result = await client.fetchInteractions({ locationId: 16808, first: 50 })
for (const review of result.interactions) {
  if (review.rating >= 4 && !review.responded) {
    await client.respondToReview(review.uuid, `Thanks for the ${review.rating}-star review!`)
  }
}
console.log("Done replying to reviews")

Auto-reply to reviews

Fetches recent reviews, filters by rating and response status, and posts replies using your template.

# Dry-run first to preview

    min_rating=4,
    only_unanswered=True,
    dry_run=True,
)

# Then send for real
results = client.workflows.auto_reply_to_reviews(
    16808,
    template="Thanks for the {rating}-star review!",
    min_rating=4,
)
print(f"Replied to {len(results)} reviews")
// Workflows are available in the Synup SDK.
// Node.js support coming soon.
ParameterTypeDescription
location_idstr | intLocation to process.
templatestrReply text. Use {rating} as placeholder.
min_ratingintOnly reply to reviews with this rating or higher. Default: 4
only_unansweredboolSkip reviews that already have a response. Default: True
dry_runboolPreview without sending. Default: False

FastAPI backend

Wire the SDK into a REST API your frontend can call.

from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
import synup

app = FastAPI()
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"])
client = synup.Synup()

@app.get("/api/locations")
async def get_locations(first: int = 10):
    page = client.locations.list(first=first)
    return {"locations": [loc.to_dict() for loc in page], "has_more": page.has_more}

@app.get("/api/locations/{location_id}/reviews")
async def get_reviews(location_id: int, first: int = 20):
    page = client.reviews.list(location_id, first=first)
    return {"reviews": [r.to_dict() for r in page]}

@app.post("/api/reviews/{interaction_id}/respond")
async def respond(interaction_id: str, content: str):
    result = client.reviews.respond(interaction_id, content)
    return result.to_dict()
import express from 'express'
import { SynupClient } from 'synup-js'

const app = express()
const client = new SynupClient({ apiKey: process.env.SYNUP_API_KEY })

app.get('/api/locations', async (req, res) => {
  const result = await client.fetchAllLocations({ first: Number(req.query.first) || 10 })
  res.json(result)
})

app.get('/api/locations/:id/reviews', async (req, res) => {
  const result = await client.fetchInteractions({ locationId: req.params.id, first: 20 })
  res.json(result)
})

app.listen(8000, () => console.log('Server on :8000'))

Full-stack dashboard

Complete app with backend + frontend for locations, reviews, analytics.

# server.py — Full-stack FastAPI app with static frontend
from fastapi import FastAPI
from fastapi.staticfiles import StaticFiles
import synup

app = FastAPI()
client = synup.Synup()

@app.get("/api/locations")
async def locations(first: int = 10):
    page = client.locations.list(first=first)
    return [loc.to_dict() for loc in page]

@app.get("/api/locations/{loc_id}/dashboard")
async def dashboard(loc_id: int):
    reviews = client.reviews.analytics.overview(loc_id)
    listings = client.listings.premium(loc_id)
    google = client.analytics.google(loc_id)
    return {
        "reviews": reviews.to_dict(),
        "listings": [l.to_dict() for l in listings],
        "google": google.to_dict(),
    }

app.mount("/", StaticFiles(directory="static", html=True))
# Run: SYNUP_API_KEY='key' uvicorn server:app --reload --port 8000
import express from 'express'
import { SynupClient } from 'synup-js'

const app = express()
const client = new SynupClient({ apiKey: process.env.SYNUP_API_KEY })

app.get('/api/locations', async (req, res) => {
  const result = await client.fetchAllLocations({ first: 10 })
  res.json(result.locations)
})

app.get('/api/locations/:id/dashboard', async (req, res) => {
  const [reviews, listings, google] = await Promise.all([
    client.fetchReviewAnalyticsOverview(req.params.id),
    client.fetchPremiumListings(req.params.id),
    client.fetchGoogleAnalytics(req.params.id),
  ])
  res.json({ reviews, listings, google })
})

app.use(express.static('public'))
app.listen(8000)
// Run: SYNUP_API_KEY='key' node server.js

Error handling

The SDK raises specific exceptions so you can handle each failure mode differently.

import synup

client = synup.Synup()

try:
    client.locations.list()
except synup.AuthenticationError:
    print("Invalid API key")
except synup.RateLimitError as e:
    print(f"Rate limited — retry after {e.retry_after}s")
except synup.NotFoundError:
    print("Resource not found")
except synup.ValidationError as e:
    print(f"Bad request: {e.response_body}")
except synup.APIConnectionError:
    print("Network error")
import { SynupClient, SynupAPIError } from 'synup-js'

const client = new SynupClient({ apiKey: "..." })

try {
  await client.fetchAllLocations()
} catch (err) {
  if (err instanceof SynupAPIError) {
    console.log(err.statusCode, err.responseBody)
  }
}

PermissionDeniedError403Insufficient permissions
NotFoundError404Resource not found
ValidationError400/422Invalid request parameters
RateLimitError429Too many requests. Has retry_after attribute.
InternalServerError5xxServer-side error. Safe to retry.
APIConnectionError---Network failure
